let snake = [{ x: 5, y: 5 }]; let direction = { x: 0, y: 0 }; let food = { x: 10, y: 10 }; let gridSize = 20; let highScore = 0; let currentScore = 0; let gameStarted = false; // Game waits for player input function startGame() { snake = [{ x: 5, y: 5 }]; direction = { x: 0, y: 0 }; food = { x: Math.floor(Math.random() * gridSize), y: Math.floor(Math.random() * gridSize) }; currentScore = 0; gameStarted = false; draw(); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Draw food ctx.fillStyle = "red"; ctx.fillRect(food.x * gridSize, food.y * gridSize, gridSize, gridSize); // Draw snake ctx.fillStyle = "lime"; snake.forEach(part => { ctx.fillRect(part.x * gridSize, part.y * gridSize, gridSize, gridSize); }); // Draw scores document.getElementById("current-score").textContent = currentScore; document.getElementById("high-score").textContent = highScore; } function update() { if (!gameStarted) return; // Wait until player swipes const head = { x: snake[0].x + direction.x, y: snake[0].y + direction.y }; // Check collisions with walls if (head.x < 0 || head.x >= canvas.width / gridSize || head.y < 0 || head.y >= canvas.height / gridSize) { startGame(); return; } // Check collisions with self if (snake.some(part => part.x === head.x && part.y === head.y)) { startGame(); return; } // Add new head to the snake snake.unshift(head); // Check if eating food if (head.x === food.x && head.y === food.y) { currentScore++; if (currentScore > highScore) highScore = currentScore; food = { x: Math.floor(Math.random() * gridSize), y: Math.floor(Math.random() * gridSize) }; } else { snake.pop(); // Remove tail if no food is eaten } draw(); } function changeDirection(event) { if (!gameStarted) { gameStarted = true; // Start game on first swipe } const swipeDirection = event.detail; switch (swipeDirection) { case "up": if (direction.y === 0) direction = { x: 0, y: -1 }; break; case "down": if (direction.y === 0) direction = { x: 0, y: 1 }; break; case "left": if (direction.x === 0) direction = { x: -1, y: 0 }; break; case "right": if (direction.x === 0) direction = { x: 1, y: 0 }; break; } } document.addEventListener("swipe", changeDirection); setInterval(update, 150); startGame();
